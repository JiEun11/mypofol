### tutorial I: 간단 CRUD 연습 (mongosh)

```javascript
test> db   					// db 전역 변수 출력
test> use mydb				// db 전역변수는 생기지만 database는 생성되지 않음
mydb> show dbs				// list database
mydb> db					// db 전역 변수 확인
mydb> show collections		// database가 생성되지 않았지만,
                            // 아직 collection도 생성되기 전!
```

```javascript
mydb> db.account.insertOne({name: 'dooly', email:'dooly@gmail.com', password:'1234'})                           // document 생성
mydb> show collections                                                                                          // collection이 생겼음!
mydb> show dbs                                                                                                  // database도 생겼음!

mydb> db.account.find()			                                                                                // document 전체 검색

mydb> db.account.insertOne({name: 'michol', email:'michol@gmail.com', password:'1234'})			                // document 생성
mydb> db.account.countDocuments()                                                                               // document counting

mydb> db.account.find({name: 'dooly'})                                                                          // document 검색, query selector 사용I
mydb> db.account.findOne({email: {$regex: /^.*gmail.*$/}})                                                      // document 검색, query selector 사용II
mydb> db.account.find({email: /^.*gmail.*$/})                                                                   // document 검색, query selector 사용III

mydb> db.account.update({name: 'dooly'}, {$set: {email: 'dooly@naver.com'}})                                    // document 업데이트I
mydb> db.account.updateOne({name: 'dooly'}, {$set: {email: 'dooly@mongo.com', status: 'backend developer'}})    // document 업데이트II
mydb> db.account.updateMany({}, {$unset: {status: 1}})                                                          // document 업데이트III, drop field

mydb> db.account.deleteOne({'email': /^.*mongo.*$/})                                                            // document 삭제
mydb> db.account.deleteMany({})                                                                                 // document 전체 삭제

mydb> show collections
mydb> db.account.drop()
mydb> show collections

```
